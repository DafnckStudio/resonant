  activeTab: 'people' | 'link' | 'embed' | 'publish';
  resource: {
    id: string;
    type: 'workflow' | 'template' | 'folder';
    name: string;
    owner: User;
  };

  // People tab
  sharedWith: SharedUser[];
  pendingInvites: string[];
  generalAccess: 'restricted' | 'workspace' | 'public';

  // Link tab
  linkEnabled: boolean;
  linkUrl: string;
  linkSettings: {
    permission: PermissionLevel;
    expiration: 'never' | '24h' | '7d' | '30d' | 'custom';
    expirationDate?: Date;
    password?: string;
    allowedDomains: string[];
    maxViews?: number;
    requireSignIn: boolean;
    allowDownload: boolean;
    showCreator: boolean;
    enableComments: boolean;
  };

  // Embed tab
  embedType: 'run' | 'view' | 'full';
  embedSettings: {
    theme: 'light' | 'dark' | 'system';
    width: string;
    height: string;
    showBranding: boolean;
    autoSubmit: boolean;
    showSuccess: boolean;
  };
  embedCode: string;

  // Publish tab
  publishType: 'free' | 'premium' | 'run-only';
  publishSettings: {
    price?: number;
    category: string;
    tags: string[];
    description: string;
    previewImage?: string;
  };
}

interface SharedUser {
  id: string;
  email: string;
  name?: string;
  avatar?: string;
  permission: PermissionLevel;
  isTeam: boolean;
  teamMembers?: number;
  addedBy: string;
  addedAt: Date;
}

type PermissionLevel = 'owner' | 'edit' | 'view' | 'execute' | 'comment';
