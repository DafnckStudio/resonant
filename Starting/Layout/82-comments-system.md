## State Management

```typescript
interface CommentsState {
  comments: Comment[];
  filter: 'all' | 'open' | 'resolved';
  searchQuery: string;
  sortBy: 'recent' | 'oldest' | 'mostReplies';
  filterOptions: {
    blockId: string | null;
    authorId: string | null;
    dateRange: 'all' | 'today' | 'week' | 'month';
    mentionsMe: boolean;
    hasAttachments: boolean;
  };
  activeThread: string | null;
  replyingTo: string | null;
  editingComment: string | null;
  draftComment: string;
}

interface Comment {
  id: string;
  workflowId: string;
  blockId?: string;
  author: User;
  content: string;
  mentions: string[];
  attachments: Attachment[];
  reactions: Reaction[];
  replies: Comment[];
  status: 'open' | 'resolved';
  resolvedBy?: User;
  resolvedAt?: Date;
  createdAt: Date;
  updatedAt?: Date;
  isEdited: boolean;
}

interface Reaction {
  emoji: string;
  users: User[];
  count: number;
}

interface Attachment {
  id: string;
  type: 'image' | 'file';
  name: string;
  url: string;
  size: number;
  mimeType: string;
}
```
