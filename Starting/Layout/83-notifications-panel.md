## State Management

```typescript
interface NotificationsState {
  notifications: Notification[];
  unreadCount: number;
  filter: {
    category: 'all' | 'mentions' | 'workflows' | 'team' | 'system';
    status: 'all' | 'unread' | 'read';
    type?: NotificationType;
    dateRange?: 'today' | 'week' | 'month' | 'all';
  };
  selectedNotifications: string[];
  preferences: NotificationPreferences;
  isLoading: boolean;
  hasMore: boolean;
}

interface Notification {
  id: string;
  type: NotificationType;
  title: string;
  body: string;
  metadata: Record<string, any>;
  isRead: boolean;
  isMuted: boolean;
  createdAt: Date;
  readAt?: Date;
  actions?: NotificationAction[];
}

type NotificationType =
  | 'mention'
  | 'comment'
  | 'workflow_success'
  | 'workflow_failed'
  | 'workflow_paused'
  | 'human_input'
  | 'shared'
  | 'access_request'
  | 'team_invite'
  | 'role_change'
  | 'integration_alert'
  | 'usage_alert'
  | 'security_alert'
  | 'product_update';

interface NotificationAction {
  label: string;
  type: 'primary' | 'secondary' | 'danger';
  action: string;
  href?: string;
}

interface NotificationPreferences {
  channels: {
    inApp: boolean;
    email: boolean;
    slack: boolean;
    push: boolean;
  };
  types: {
    [key in NotificationType]: {
      inApp: boolean;
      email: boolean;
      slack: boolean;
      push: boolean;
    };
  };
  quietHours: {
    enabled: boolean;
    start: string;
    end: string;
    allowCritical: boolean;
  };
}
```
