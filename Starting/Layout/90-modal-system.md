```
1. On open:
   - Focus first focusable element (usually close button or first input)
   - Trap focus within modal (Tab cycles through modal elements only)

2. On close:
   - Return focus to element that triggered modal open

3. Focus order:
   Close button â†’ Header elements â†’ Content â†’ Footer buttons
```

## State Management

```typescript
// Modal state interface
interface ModalState {
  isOpen: boolean;
  type: 'dialog' | 'sheet' | 'alert' | 'wizard';
  size: 'sm' | 'md' | 'lg' | 'xl' | 'full';
  position: 'center' | 'right' | 'bottom';
  dismissible: boolean;
  data?: any;
  onConfirm?: () => void;
  onCancel?: () => void;
}

// Modal actions
type ModalAction =
  | { type: 'OPEN'; payload: Partial<ModalState> }
  | { type: 'CLOSE' }
  | { type: 'UPDATE'; payload: Partial<ModalState> }
  | { type: 'NEXT_STEP' }
  | { type: 'PREV_STEP' };

// Modal context values
const ModalContext = {
  openModal: (config: Partial<ModalState>) => void,
  closeModal: () => void,
  confirm: (options: ConfirmOptions) => Promise<boolean>,
  alert: (options: AlertOptions) => Promise<void>,
  prompt: (options: PromptOptions) => Promise<string | null>,
};
```
