## State Management

```typescript
interface TeamManagementState {
  activeTab: 'members' | 'teams' | 'roles' | 'pending' | 'activity';
  members: Member[];
  teams: Team[];
  roles: Role[];
  pendingInvitations: Invitation[];
  activityLog: ActivityEntry[];
  selectedMembers: string[];
  filters: {
    role: string | null;
    team: string | null;
    status: 'active' | 'inactive' | 'all';
    searchQuery: string;
  };
  modals: {
    invite: boolean;
    createTeam: boolean;
    editTeam: Team | null;
    createRole: boolean;
    editRole: Role | null;
    removeMember: Member | null;
    transferOwnership: boolean;
    memberProfile: Member | null;
  };
  pagination: {
    page: number;
    perPage: number;
    total: number;
  };
}

interface Member {
  id: string;
  email: string;
  name: string;
  avatar?: string;
  role: Role;
  teams: Team[];
  status: 'online' | 'offline' | 'away';
  lastActive: Date;
  joinedAt: Date;
  workflowCount: number;
  executionCount: number;
}

interface Team {
  id: string;
  name: string;
  icon: string;
  description?: string;
  members: string[];
  workflowCount: number;
  settings: TeamSettings;
  createdAt: Date;
  createdBy: string;
}

interface Role {
  id: string;
  name: string;
  icon: string;
  description: string;
  permissions: Permission[];
  isSystem: boolean;
  memberCount: number;
}

interface Invitation {
  id: string;
  email: string;
  role: Role;
  team?: Team;
  invitedBy: Member;
  sentAt: Date;
  expiresAt: Date;
  status: 'pending' | 'expiring' | 'expired';
}
```
